Tool Calls SSE Stream Analysis

问题诊断

根本原因：后端生成的tool_calls流式数据格式不符合OpenAI标准规范

当前数据格式问题：

Chunk序号 Index ID Name Arguments 问题
1 0 call_d64f4737... ssh_exec { arguments不完整
2-10 1 tc-1783323... - 片段拼接 使用了不同的index和ID！
最后(finish) 0 & 1 两个ID - 未完整合并 两个独立的tool_call对象

正确的OpenAI标准格式：

// 所有chunks应该使用相同的index和ID！

Chunk 1:
{
  "tool_calls": [{
    "index": 0,
    "id": "call_d64f4737...",
    "type": "function",
    "function": {
      "name": "ssh_exec",
      "arguments": "{"
    }
  }]
}

Chunk 2:
{
  "tool_calls": [{
    "index": 0,  // 相同的index！
    "id": "call_d64f4737...",  // 相同的ID！
    "type": "function",
    "function": {
      "arguments": "\"command\": \"free -h\", "
    }
  }]
}

Chunk 3:
{
  "tool_calls": [{
    "index": 0,  // 相同的index！
    "id": "call_d64f4737...",  // 相同的ID！
    "type": "function",
    "function": {
      "arguments": "\"host\": \"192.168.100.186\"}"
    }
  }]
}

Final chunk:
{
  "tool_calls": [{
    "index": 0,
    "id": "call_d64f4737...",
    "type": "function",
    "function": {
      "name": "ssh_exec",
      "arguments": "{\"command\": \"free -h\", \"host\": \"192.168.100.186\"}"
    }
  }],
  "finish_reason": "tool_calls"
}

前端合并逻辑分析

tiny-robot-kit使用以下算法合并tool_calls：

function mergeToolCalls(existing, incoming) {
  // 1. 创建Map，以index为键
  let existingMap = new Map(existing.map(tc => [tc.index, tc]));
  let incomingMap = new Map(incoming.map(tc => [tc.index, tc]));
  
  // 2. 对于新数组中的每个元素
  for (let [index, newTc] of incomingMap) {
    if (existingMap.has(index)) {
      // 3. 如果存在相同index，则合并
      let oldTc = existingMap.get(index);
      existingMap.set(index, deepMerge(oldTc, newTc));
    } else {
      // 4. 否则添加新元素
      existingMap.set(index, newTc);
    }
  }
  
  // 5. 返回合并后的数组
  return Array.from(existingMap.values());
}
问题：由于后端使用了index 0和index 1两个不同的index， 前端会将它们视为两个不同的工具调用，而不是同一个工具调用的分片！

解决方案

方案1：修改后端（推荐）

修改后端生成SSE流的代码，确保：

  1. 所有tool_calls分片使用相同的index（通常为0）
  2. 所有tool_calls分片使用相同的ID
  3. 第一个chunk包含完整的name
  4. 后续chunks只包含arguments的增量
  5. 最后的finish chunk包含完整的合并结果

方案2：修改前端（临时方案）

如果暂时无法修改后端，可以在前端添加自定义处理逻辑：

// 在responseProvider中添加transform函数
responseProvider: async (requestBody, abortSignal) => {
  const response = await fetch('/api/v1/chat/messages/stream/events', {
    method: 'POST',
    headers: { ... },
    body: JSON.stringify({ ... }),
    signal: abortSignal
  });

  if (!response.ok) throw new Error('Request failed');

  // 自定义SSE流处理
  return customSSEToGenerator(response);
}

// 自定义SSE流解析器
async function* customSSEToGenerator(response) {
  const reader = response.body.getReader();
  const decoder = new TextDecoder();
  let buffer = '';
  
  // 用于合并tool_calls
  const toolCallsBuffer = new Map();
  
  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    
    buffer += decoder.decode(value, { stream: true });
    const lines = buffer.split('\n');
    buffer = lines.pop() || '';
    
    for (const line of lines) {
      if (!line.trim() || !line.startsWith('data: ')) continue;
      
      const data = line.slice(6);
      if (data === '[DONE]') return;
      
      try {
        const chunk = JSON.parse(data);
        
        // 自定义处理tool_calls
        if (chunk.choices?.[0]?.delta?.tool_calls) {
          const deltaTc = chunk.choices[0].delta.tool_calls;
          
          // 合并逻辑：优先使用第一个完整出现的ID和name
          for (const tc of deltaTc) {
            const existingTc = toolCallsBuffer.get(tc.id);
            
            if (existingTc) {
              // 合并arguments
              existingTc.function.arguments += tc.function.arguments || '';
            } else {
              // 新的工具调用
              toolCallsBuffer.set(tc.id, {
                index: 0, // 强制使用相同的index！
                id: tc.id,
                type: 'function',
                function: {
                  name: tc.function?.name || '',
                  arguments: tc.function?.arguments || ''
                }
              });
            }
          }
          
          // 发送合并后的tool_calls（使用统一的index）
          chunk.choices[0].delta.tool_calls = Array.from(toolCallsBuffer.values())
            .map((tc, idx) => ({ ...tc, index: idx }));
        }
        
        yield chunk;
      } catch (e) {
        console.warn('Failed to parse SSE data:', data, e);
      }
    }
  }
}

// 所有chunks - index 0

// 最后finish - 一个完整的对象

  {"index": 0, "id": "call_abc", "function": {"name": "ssh_exec", "arguments": "完整JSON"}}
